-- Logic variables
local tweenEnabled = false
local instantEnabled = false
local lockedTarget = nil
local savedCFrame = nil
local connection

-- Team & target check
local function isValidTarget(plr)
	if plr == LocalPlayer then return false end
	if not plr.Character then return false end

	local humanoid = plr.Character:FindFirstChild("Humanoid")
	local root = plr.Character:FindFirstChild("HumanoidRootPart")
	if not humanoid or not root then return false end
	if humanoid.Health <= 0 then return false end

	local myTeam = tostring(LocalPlayer.Team)
	local theirTeam = tostring(plr.Team)

	if string.find(myTeam,"Marine") then
		-- Marines can only target Pirates
		return string.find(theirTeam,"Pirate") ~= nil
	else
		return true -- Pirates can target anyone
	end
end

local function getNearestPlayer()
	local char = LocalPlayer.Character
	if not char then return end
	local root = char:FindFirstChild("HumanoidRootPart")
	if not root then return end

	local closest, dist = nil, math.huge
	for _, plr in pairs(Players:GetPlayers()) do
		if isValidTarget(plr) then
			local targetRoot = plr.Character.HumanoidRootPart
			local mag = (targetRoot.Position - root.Position).Magnitude
			if mag < dist then
				dist = mag
				closest = plr
			end
		end
	end
	return closest, dist
end

-- Prevent falling
local function stabilize(root)
	root.AssemblyLinearVelocity = Vector3.zero
	root.AssemblyAngularVelocity = Vector3.zero
end

-- Update Stud Indicators
task.spawn(function()
	while true do
		local target, dist = getNearestPlayer()
		local studs = target and math.floor(dist) or 0
		TweenButton.Text = "Tween ["..(tweenEnabled and "ON" or "OFF").."] ["..studs.." Studs]"
		InstantButton.Text = "Instant ["..(instantEnabled and "ON" or "OFF").."] ["..studs.." Studs]"
		task.wait(0.1)
	end
end)

-- TWEEN TP
TweenButton.MouseButton1Click:Connect(function()
	tweenEnabled = not tweenEnabled
	instantEnabled = false
	if connection then connection:Disconnect() end

	if tweenEnabled then
		lockedTarget = getNearestPlayer()
		connection = RunService.Heartbeat:Connect(function()
			local char = LocalPlayer.Character
			if not char then return end
			local root = char:FindFirstChild("HumanoidRootPart")
			local humanoid = char:FindFirstChild("Humanoid")
			if not root or not humanoid then return end

			-- Auto retarget if invalid
			if not lockedTarget or not isValidTarget(lockedTarget) then
				lockedTarget = getNearestPlayer()
				if not lockedTarget then return end
			end

			local targetRoot = lockedTarget.Character:FindFirstChild("HumanoidRootPart")
			if not targetRoot then return end

			local goal = targetRoot.CFrame * CFrame.new(0,0,3)
			local dist = (root.Position - goal.Position).Magnitude
			local time = dist / 350

			stabilize(root)
			humanoid:ChangeState(Enum.HumanoidStateType.Physics)

			TweenService:Create(root, TweenInfo.new(time, Enum.EasingStyle.Linear), {CFrame = goal}):Play()
		end)
	else
		lockedTarget = nil
	end
end)

-- INSTANT TP
InstantButton.MouseButton1Click:Connect(function()
	instantEnabled = not instantEnabled
	tweenEnabled = false
	if connection then connection:Disconnect() end

	local char = LocalPlayer.Character
	if not char then return end
	local root = char:FindFirstChild("HumanoidRootPart")
	local humanoid = char:FindFirstChild("Humanoid")
	if not root or not humanoid then return end

	if instantEnabled then
		lockedTarget = getNearestPlayer()
		savedCFrame = root.CFrame
		connection = RunService.Heartbeat:Connect(function()
			-- Auto retarget
			if not lockedTarget or not isValidTarget(lockedTarget) then
				lockedTarget = getNearestPlayer()
				if not lockedTarget then return end
			end

			local targetRoot = lockedTarget.Character:FindFirstChild("HumanoidRootPart")
			if not targetRoot then return end

			stabilize(root)
			humanoid:ChangeState(Enum.HumanoidStateType.Physics)

			root.CFrame = targetRoot.CFrame * CFrame.new(0,0,3)
		end)
	else
		if savedCFrame then
			root.CFrame = savedCFrame
		end
		lockedTarget = nil
	end
end)
